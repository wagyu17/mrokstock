```python
def fibonacci(n):
    fib_seq = [0, 1]  # 初期の2つの要素
    for i in range(2, n):
        fib_seq.append(fib_seq[i-1] + fib_seq[i-2])  # 前の2つの要素の和を次の要素として追加
    return fib_seq[:n]

n = int(input("フィボナッチ数列の長さを入力してください: "))
print(fibonacci(n))
```
