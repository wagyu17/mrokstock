T_{11}{\prime} = \cos^2\theta T_{11} + (\cos\theta)(\sin\theta) T_{12} + (\cos\theta)(\sin\theta) T_{21} + \sin^2\theta T_{22}

T_{22}{\prime} = -(\cos\theta)(\sin\theta) T_{11} + \cos^2\theta T_{12} - \sin^2\theta T_{21} + (\cos\theta)(\sin\theta) T_{22}

T_{12}{\prime} = \sin^2\theta T_{11} - (\cos\theta)(\sin\theta) T_{12} - (\cos\theta)(\sin\theta) T_{21} + \cos^2\theta T_{22}
