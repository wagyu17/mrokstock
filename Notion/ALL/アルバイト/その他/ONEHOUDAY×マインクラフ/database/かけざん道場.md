```javascript
// 変数の準備
let current_input = 0
let ans = 0
let height = 0
let width = 0

// 1. 「mondai」コマンド：問題作成とエージェント建築
player.onChat("mondai", function () {
    // リセット処理
    current_input = 0
    player.execute("title @a actionbar §f現在の答え: 0")

    // 乱数生成（1～9）
    let num1 = Math.randomRange(1, 9)
    let num2 = Math.randomRange(1, 9)
    
    // 答えとサイズを記録
    ans = num1 * num2
    height = num1
    width = num2

    // 画面表示
    player.execute("title @a title §eもんだい！")
    player.execute("title @a subtitle §b" + num1 + " × " + num2 + " ＝ ？")

    // エージェント設定
    agent.teleportToPlayer()
    agent.move(FORWARD, 3)
    
    // ★変更点1：スロット1に「ダイヤモンドブロック」をセット
    agent.setItem(DIAMOND_BLOCK, 64, 1)
    agent.setSlot(1)

    // 建築ループ（縦 x 横）
    for (let i = 0; i < num1; i++) {
        for (let j = 0; j < num2; j++) {
            agent.place(FORWARD)
            agent.move(RIGHT, 1)
        }
        // 折り返し
        agent.move(LEFT, num2)
        agent.move(UP, 1)
    }

    // 邪魔にならないよう退避
    agent.move(UP, 2)

    // アイテム配布（リモコン用）
    mobs.give(mobs.target(LOCAL_PLAYER), APPLE, 1)         // +1
    mobs.give(mobs.target(LOCAL_PLAYER), GOLDEN_APPLE, 1)  // +10
    mobs.give(mobs.target(LOCAL_PLAYER), STICK, 1)         // 決定
    // ★変更点2：リセット用に「羽根」を配る
    mobs.give(mobs.target(LOCAL_PLAYER), FEATHER, 1)       // リセット
    
    // 操作説明を表示
    player.execute("title @a actionbar §cリンゴ=+1, §e金=+10, §f棒=決定, §b羽根=リセット")
})

// 2. リンゴを使った時 (+1)
player.onItemInteracted(APPLE, function () {
    current_input += 1
    player.execute("title @a actionbar §a現在の答え: " + current_input)
})

// 3. 金のリンゴを使った時 (+10)
player.onItemInteracted(GOLDEN_APPLE, function () {
    current_input += 10
    player.execute("title @a actionbar §a現在の答え: " + current_input)
})

// ★変更点3：羽根を使った時 (リセット)
player.onItemInteracted(FEATHER, function () {
    current_input = 0
    player.execute("title @a actionbar §cリセットしました: 0")
})

// 4. 棒を使った時 (決定・答え合わせ)
player.onItemInteracted(STICK, function () {
    if (current_input == ans) {
        // 正解の演出
        player.execute("title @a title §6だいせいかい！！")
        mobs.spawn(FIREWORKS_ROCKET, player.position().add(pos(0, 5, 0)))

        // ★お片付け機能（破壊して回収）
        // エージェントを建築開始位置に戻す
        agent.move(DOWN, 2)       
        agent.move(DOWN, height)  

        // ブロックを壊して回るループ
        for (let i = 0; i < height; i++) {
            for (let j = 0; j < width; j++) {
                agent.destroy(FORWARD) // 壊す
                agent.collectAll()     // 落ちたアイテムを拾う
                agent.move(RIGHT, 1)
            }
            agent.move(LEFT, width)
            agent.move(UP, 1)
        }
        
        // 終わったら退避
        agent.move(UP, 2)

    } else {
        // 不正解の演出
        player.execute("title @a title §cざんねん...")
        player.execute("title @a subtitle §e正解は " + ans + " だよ")
        
        // 入力を0に戻す
        current_input = 0
        player.execute("title @a actionbar §a現在の答え: 0")
    }
})
```
